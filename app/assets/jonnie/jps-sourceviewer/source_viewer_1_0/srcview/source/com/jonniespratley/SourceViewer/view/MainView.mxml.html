    <mx:Script>
        <![CDATA[
            import com.jonniespratley.SourceViewer.events.GetDirectoryEvent;
            import com.jonniespratley.SourceViewer.events.SelectedFileEvent;
            import com.jonniespratley.SourceViewer.model.ModelLocator;
            import com.jonniespratley.SourceViewer.utils.ByteUtils;
            import com.jonniespratley.SourceViewer.vo.FileVO;

            import mx.collections.ArrayCollection;
            import mx.controls.Alert;
            import mx.controls.textClasses.TextRange;
            import mx.managers.SystemManager;
            import mx.utils.StringUtil;

            [Bindable]private var fileVO:FileVO;

            [Bindable]private var model:ModelLocator=ModelLocator.getInstance();

            private var utils:ByteUtils=new ByteUtils();

            private var timer:Timer;

            private var timerWait:Number=10;

            private var fileToDownload:FileReference;

            private var urlRequest:URLRequest;

            private function init():void
            {
                getDirectory(txt_dir.text);
            }

            private function tree_labelFunc(item:Object):String
            {
                var label:String=item.label;

                if (code_tree.dataDescriptor.hasChildren(item))
                {
                    label+=" (" + code_tree.dataDescriptor.getChildren(item).length + ")";
                }
                return label;
            }

            private function getDirectory(dir:String):void
            {
                var e:GetDirectoryEvent=new GetDirectoryEvent(dir);
                e.dispatch();
            }

            private function onSelect(event:Event):void
            {
                if (event.currentTarget.selectedItem is Array)
                {
                    //Fuck
                    trace('Folder');
                }
                else
                {

                    fileVO=event.currentTarget.selectedItem as FileVO;
                    var e_selectedFile:SelectedFileEvent=new SelectedFileEvent(fileVO);
                    e_selectedFile.dispatch();

                    if (fileVO.aExt == 'png' || fileVO.aExt == 'swf')
                    {
                        vs_main.selectedIndex=1;
                    }
                    else
                    {
                        vs_main.selectedIndex=0;
                    }
                }
            }

            private function downloadFile(whatFile:String):void
            {
                urlRequest=new URLRequest(whatFile);
                fileToDownload=new FileReference();

                fileToDownload.addEventListener(ProgressEvent.PROGRESS, onDownloadProgress);
                fileToDownload.addEventListener(Event.COMPLETE, onDownloadComplete);
                fileToDownload.addEventListener(SecurityErrorEvent.SECURITY_ERROR, onSecurityError);
                fileToDownload.addEventListener(IOErrorEvent.IO_ERROR, onIOError);
                fileToDownload.addEventListener(Event.CANCEL, onDownloadCancel);

                try
                {
                    fileToDownload.download(urlRequest);
                    timer=new Timer(timerWait);
                    timer.addEventListener(TimerEvent.TIMER_COMPLETE, onTimerComplete);
                    timer.start();
                }

                catch (error:Error)
                {
                    Alert.show("Unable to download file.\n Let Jonnie know that his SourceViewer is screwed up, and tell him to hurry up and fix it.", "Error");

                }
            }

            private function onDownloadProgress(event:ProgressEvent):void
            {
                var total:int=event.bytesTotal;
                var loaded:int=event.bytesLoaded;

                download_progress.visible=true;
                download_progress.includeInLayout=true;
                download_progress.label='DOWNLOADED ' + utils.format(loaded, ByteUtils.KBYTES, 0) + ' OF ' + utils.format(total, ByteUtils.KBYTES, 0);
                download_progress.setProgress(loaded, total);
            }

            private function onDownloadComplete(event:Event):void
            {
                hideProgress();
            }

            private function hideProgress():void
            {
                download_progress.label='';
                download_progress.includeInLayout=false;
                download_progress.visible=false;
            }

            private function onSecurityError(event:SecurityErrorEvent):void
            {
                Alert.show("Security Error.\n Let Jonnie know that his SourceViewer is screwed up, and tell him to hurry up and fix it.", "Error");
                hideProgress();
            }

            private function onIOError(event:IOErrorEvent):void
            {
                Alert.show("IO Error.\n Let Jonnie know that his SourceViewer is screwed up, and tell him to hurry up and fix it.", "Error");
                hideProgress();
            }

            private function onDownloadCancel(event:Event):void
            {
                //Alert.show( "Download Cancelled.\n Let Jonnie know that his SourceViewer is screwed up, and tell him to hurry up and fix it.", "Oh Shit!" );
                hideProgress();
            }

            private function onTimerComplete(event:TimerEvent):void
            {
                timer.stop();
                hideProgress();
            }

            private function copyCode():void
            {
                System.setClipboard(codeView._sourceCode);
            }
        ]]>
