<?xml version="1.0" encoding="utf-8"?>
<mx:VBox
    width="100%"
    height="100%"
    styleName="padding"
    creationComplete="init()"
    xmlns:mx="http://www.adobe.com/2006/mxml">
    <mx:Script>
        <![CDATA[
            import com.jonniespratley.SourceViewer.events.GetDirectoryEvent;
            import com.jonniespratley.SourceViewer.vo.FileVO;
            import com.jonniespratley.SourceViewer.events.SelectedFileEvent;
            import com.jonniespratley.SourceViewer.model.ModelLocator;
            [Bindable] private var model:ModelLocator = ModelLocator.getInstance();

            private function init():void
            {
                getDirectory();
            }

            private function tree_labelFunc( item:Object ):String
            {
                var label:String = item.label;

                if ( code_tree.dataDescriptor.hasChildren( item ) )
                {
                    label += " (" + code_tree.dataDescriptor.getChildren( item ).length + ")";
                }
                return label;
            }

            private function getDirectory():void
            {
                var e:GetDirectoryEvent = new GetDirectoryEvent();
                    e.dispatch();
            }

            private function onSelect( event:Event ):void
            {
                if ( event.currentTarget.selectedItem is Array )
                {
                    return;
                }
                var vo:FileVO = event.currentTarget.selectedItem as FileVO;
                var e_selectedFile:SelectedFileEvent = new SelectedFileEvent( vo );
                    e_selectedFile.dispatch();
            }

        ]]>
    </mx:Script>
      <mx:Box width="100%" styleName="ccControlBar" >
            <mx:Label text="Code Tree"/>
      </mx:Box>
        <mx:Box width="100%" height="100%" styleName="ccControlBar" backgroundAlpha="1" >
            <mx:filters>
                  <flash.filters:BevelFilter xmlns:flash.filters="flash.filters.*" angle="90" blurX="1" blurY="1"
                       distance="1" highlightAlpha=".4" shadowAlpha=".4" highlightColor="0x000000" shadowColor="0xFFFFFF" knockout="true" />
              </mx:filters>
        <mx:Tree id="code_tree"
            width="100%"
            height="100%"
            change="onSelect( event )"
            labelFunction="tree_labelFunc"
            dataProvider="{ model.directoryCollection }"/>
        <mx:HBox width="100%">
            <mx:LinkButton
                click="getDirectory()"
                label="Refresh"/>
            <mx:LinkButton
                label="Download"/>
        </mx:HBox>
    </mx:Box>
</mx:VBox>
