package com.jonniespratley.SourceViewer.commands
{
    import com.adobe.cairngorm.commands.ICommand;
    import com.adobe.cairngorm.control.CairngormEvent;
    import com.jonniespratley.SourceViewer.business.ReadFileDelegate;
    import com.jonniespratley.SourceViewer.events.SelectedFileEvent;
    import com.jonniespratley.SourceViewer.model.ModelLocator;
    import com.jonniespratley.SourceViewer.vo.*;
    
    import mx.controls.Alert;
    import mx.rpc.IResponder;
    import mx.rpc.events.FaultEvent;
    import mx.rpc.events.ResultEvent;

    /**
     * For every feature or use case in your application, a corresponding custom command
     * and event must be created. Each command contains an execute() method that is run
     * by the framework when the user action has been executed.
     * These classes are occasionally referred to as worker classes because they carry
     * out the work for an application.
     *
     */
    public class SelectedFileCommand implements ICommand, IResponder
    {
        private var model:ModelLocator = ModelLocator.getInstance();

        public function execute( event:CairngormEvent ) : void
        {
            var evt:SelectedFileEvent = event as SelectedFileEvent;
            
            model.selectedFile = evt.aFile;
            
            var delegate:ReadFileDelegate = new ReadFileDelegate( this );
                delegate.readFile( evt.aFile );

        }

        public function result( data:Object ) : void
        {
            var result:ResultEvent = data as ResultEvent;
            model.fileContents = data.result as String;
            
        }

        public function fault( fault:Object ):void
        {
            var faultEvt:FaultEvent = fault as FaultEvent;
            Alert.show(faultEvt.fault.toString(), "Error");
            trace(faultEvt.fault.faultString);
        }

    }
}
