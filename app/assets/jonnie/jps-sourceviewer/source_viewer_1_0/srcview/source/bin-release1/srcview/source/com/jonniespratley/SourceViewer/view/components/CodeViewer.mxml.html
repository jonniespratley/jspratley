    creationComplete="init()"

    <mx:Script>
        <![CDATA[
            import com.jonniespratley.SourceViewer.events.GetDirectoryEvent;
            import com.jonniespratley.SourceViewer.vo.FileVO;
            import com.jonniespratley.SourceViewer.events.SelectedFileEvent;
            import com.jonniespratley.SourceViewer.model.ModelLocator;

            import mx.collections.ArrayCollection;
            import mx.controls.textClasses.TextRange;
            import mx.utils.StringUtil;
            import mx.controls.Alert;
            import mx.managers.SystemManager;

            [Bindable] private var fileVO:FileVO;
            [Bindable] private var model:ModelLocator = ModelLocator.getInstance();
            private var fileToDownload:FileReference;
            private var urlRequest:URLRequest;

            private function init():void
            {
                getDirectory();
            }

            private function tree_labelFunc( item:Object ):String
            {
                var label:String = item.label;

                if ( code_tree.dataDescriptor.hasChildren( item ) )
                {
                    label += " (" + code_tree.dataDescriptor.getChildren( item ).length + ")";
                }
                return label;
            }

            private function getDirectory():void
            {
                var e:GetDirectoryEvent = new GetDirectoryEvent();
                    e.dispatch();
            }

            private function onSelect( event:Event ):void
            {
                if ( event.currentTarget.selectedItem is Array )
                {
                    return;
                }

                fileVO = event.currentTarget.selectedItem as FileVO;
                var e_selectedFile:SelectedFileEvent = new SelectedFileEvent( fileVO );
                    e_selectedFile.dispatch();
            }

            private function copyCode():void
            {
                var code:String = txt_code.text;

                System.setClipboard( code );

                Alert.show( 'Code is in your clipboard' );
            }


            private function downloadFile( whatFile:String ):void
            {
                fileToDownload = new FileReference();
                urlRequest = new URLRequest( 'http://' + whatFile );

                try {
                    fileToDownload.download( urlRequest);

                } catch ( error:Error )
                {
                    Alert.show( "Unable to download file." );
                    trace( error.message );
                    trace( error.name );
                    trace( error.errorID );
                    trace( error.getStackTrace() );
                }
            }


        ]]>
    </mx:Script>

                <mx:Tree id="code_tree"
                    width="100%"
                    height="100%"
                    change="onSelect( event )"
                    labelFunction="tree_labelFunc"

                    <mx:Button

                    <mx:Button
