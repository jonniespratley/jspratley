package com.jonniespratley.SourceViewer.control
{
    import com.adobe.cairngorm.control.FrontController;
    import com.jonniespratley.SourceViewer.commands.*;
    import com.jonniespratley.SourceViewer.events.*;

    /**
     * The Front Controller is a central place where all the commands are mapped to
     * the relevant events. Once an event is dispatched, the Front Controller finds
     * the corresponding command to execute.
     */
    public final class SourceViewerController extends FrontController
    {
        public function SourceViewerController()
        {
            this.initialize();
        }

        private function initialize() : void
        {
            //Map events.WHAT TYPE OF EVENT, to COMMAND
            this.addCommand( GetDirectoryEvent.GET_DIRECTORY_EVENT, GetDirectoryCommand );
            this.addCommand( ReadFileEvent.READ_FILE_EVENT, ReadFileCommand );
            this.addCommand( SelectedFileEvent.SELECTED_FILE_EVENT, SelectedFileCommand );
            this.addCommand( GetWorkspaceEvent.GET_WORKSPACE_EVENT, GetWorkspaceCommand );

            //todo: add commands
        }
    }
}
