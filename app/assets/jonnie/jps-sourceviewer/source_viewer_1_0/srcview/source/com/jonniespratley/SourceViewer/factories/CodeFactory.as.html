package com.jonniespratley.SourceViewer.factories
{
    import com.adobe.serialization.json.JSON;
    import com.jonniespratley.SourceViewer.vo.FileVO;

    import mx.collections.ArrayCollection;

    public class CodeFactory
    {
        public function CodeFactory()
        {
        }

        static public function parseJSONResult(json:Object):ArrayCollection
        {
            if (json is String)
            {
                //get the raw JSON data and cast to String
                var rawData:String=String(json);

                //decode the data to ActionScript 
                var tempArray:Array=(JSON.decode(rawData) as Array);

                    //var tempAC:ArrayCollection = new ArrayCollection( tempArray );

            }
            return new ArrayCollection(tempArray);
        }

    }
